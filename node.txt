  const input    = document.getElementById('taskInput');
  const addBtn   = document.getElementById('addBtn');
  const taskList = document.getElementById('taskList');
  
  function addTask() {
    const text = input.value.trim();   // .trim() removes extra spaces
    if (text === '') return;           // Don't add empty tasks
  
    // Create a <li> element in memory
    const li = document.createElement('li');
  
    // Set its HTML content (task text + 2 buttons)
    li.innerHTML = `
      <span>${text}</span>
      <button class="doneBtn">Done</button>
      <button class="delBtn">Delete</button>
    `;
  
    // Done button: strike through the text
    li.querySelector('.doneBtn').addEventListener('click', () => {
      li.querySelector('span').style.textDecoration = 'line-through';
    });
  
    // Delete button: remove the <li> from the page
    li.querySelector('.delBtn').addEventListener('click', () => li.remove());
  
    taskList.appendChild(li);   // Add the <li> to the <ul>
    input.value = '';           // Clear the input box
  }
  
  addBtn.addEventListener('click', addTask);
  
  // Also add task when user presses Enter
  input.addEventListener('keypress', e => { if (e.key === 'Enter') addTask(); });

  const convertBtn = document.getElementById('convertBtn');
  const resultDiv  = document.getElementById('result');
  
  async function convertCurrency() {   // async = this function can wait for data from internet
    const amount = document.getElementById('amount').value;
    const from   = document.getElementById('fromCurrency').value;
    const to     = document.getElementById('toCurrency').value;
  
    resultDiv.innerText = 'Fetching rates...';
  
    try {
      // fetch() sends a request to the internet and waits for the reply
      const response = await fetch(`https://api.exchangerate-api.com/v4/latest/${from}`);
      const data     = await response.json();   // Convert reply to a JavaScript object
  
      const rate            = data.rates[to];          // Get exchange rate
      const convertedAmount = (amount * rate).toFixed(2); // Multiply and round to 2 decimals
  
      resultDiv.innerText = `${amount} ${from} = ${convertedAmount} ${to}`;
    } catch (error) {
      // If internet fails or API is down, show a friendly message
      resultDiv.innerText = 'Error fetching rates. Try again.';
    }
  }
  
  convertBtn.addEventListener('click', convertCurrency);

import { useState } from "react";

function RegistrationForm() {
  const [formData, setFormData] = useState({
    name: "",
    email: "",
    password: ""
  });

  const [errors, setErrors] = useState({});
  const [submittedData, setSubmittedData] = useState(null);

  const handleChange = (e) => {
    setFormData({
      ...formData,
      [e.target.name]: e.target.value
    });
  };

  const validate = () => {
    let newErrors = {};

    if (formData.name.trim() === "") {
      newErrors.name = "Name is required";
    }

    const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailPattern.test(formData.email)) {
      newErrors.email = "Invalid email format";
    }

    if (formData.password.length < 8) {
      newErrors.password = "Password must be at least 8 characters";
    }

    return newErrors;
  };

  const handleSubmit = (e) => {
    e.preventDefault();

    const validationErrors = validate();
    setErrors(validationErrors);

    if (Object.keys(validationErrors).length === 0) {
      setSubmittedData(formData);
    }
  };

  return (
    <div className="container">
      <div className="form-box">
        <h2>User Registration</h2>

        <form onSubmit={handleSubmit}>
          <div>
            <label>Name:</label><br />
            <input
              type="text"
              name="name"
              value={formData.name}
              onChange={handleChange}
            />
            <p>{errors.name}</p>
          </div>

          <div>
            <label>Email:</label><br />
            <input
              type="text"
              name="email"
              value={formData.email}
              onChange={handleChange}
            />
            <p>{errors.email}</p>
          </div>

          <div>
            <label>Password:</label><br />
            <input
              type="password"
              name="password"
              value={formData.password}
              onChange={handleChange}
            />
            <p>{errors.password}</p>
          </div>

          <button type="submit">Register</button>
        </form>

        {submittedData && (
          <div>
            <h3>Registered User Details</h3>
            <p>Name: {submittedData.name}</p>
            <p>Email: {submittedData.email}</p>
            <p>Password: {submittedData.password}</p>
          </div>
        )}
      </div>
    </div>
  );
}

export default RegistrationForm;

import React, { useState, useEffect } from "react";
import "./App.css";

function App() {

  const [tasks, setTasks] = useState([]);
  const [taskText, setTaskText] = useState("");

  const addTask = () => {
    if (taskText.trim() === "") return;

    const newTask = {
      id: Date.now(),
      text: taskText,
      completed: false
    };

    setTasks([...tasks, newTask]);
    setTaskText("");
    console.log("Task Added");
  };

  const deleteTask = (id) => {
    setTasks(tasks.filter(task => task.id !== id));
    console.log("Task Deleted");
  };

  const toggleComplete = (id) => {
    setTasks(
      tasks.map(task =>
        task.id === id ? { ...task, completed: !task.completed } : task
      )
    );
  };

  useEffect(() => {
    console.log("Task List Updated");
  }, [tasks]);

  return (
    <div className="container">
      <h2>To-Do App</h2>

      <input
        type="text"
        value={taskText}
        placeholder="Enter task"
        onChange={(e) => setTaskText(e.target.value)}
      />

      <button onClick={addTask}>Add Task</button>

      <ul>
        {tasks.map(task => (
          <li key={task.id}>
            <input
              type="checkbox"
              checked={task.completed}
              onChange={() => toggleComplete(task.id)}
            />

            <span className={task.completed ? "done" : ""}>
              {task.text}
            </span>

            <button onClick={() => deleteTask(task.id)}>Delete</button>
          </li>
        ))}
      </ul>
    </div>
  );
}

export default App;

import React, { useState } from "react";

function ShoppingList() {
  const [item, setItem] = useState("");
  const [items, setItems] = useState([]);

  const addItem = () => {
    if (item === "") return;

    setItems([
      ...items,
      { id: Date.now(), name: item, bought: false }
    ]);
    setItem("");
  };

  const toggleBought = (id) => {
    setItems(
      items.map(i =>
        i.id === id ? { ...i, bought: !i.bought } : i
      )
    );
  };

  const deleteItem = (id) => {
    setItems(items.filter(i => i.id !== id));
  };

  return (
    <div>
      <h2>Shopping List</h2>

      <input
        value={item}
        onChange={(e) => setItem(e.target.value)}
        placeholder="Enter item"
      />
      <button onClick={addItem}>Add</button>

      <ul>
        {items.map(i => (
          <li key={i.id}>
            <input
              type="checkbox"
              onChange={() => toggleBought(i.id)}
            />
            <span style={{ textDecoration: i.bought ? "line-through" : "none" }}>
              {i.name}
            </span>
            <button onClick={() => deleteItem(i.id)}>Delete</button>
          </li>
        ))}
      </ul>
    </div>
  );
}

export default ShoppingList;

import { useState } from "react";

function Register() {
  const [formData, setFormData] = useState({
    username: "",
    email: "",
    password: "",
    gender: "",
    hobbies: [],
    country: "",
    about: "",
    file: null
  });

  // Handle text inputs
  const handleChange = (e) => {
    const { name, value } = e.target;

    setFormData({
      ...formData,
      [name]: value
    });
  };

  // Handle checkboxes
  const handleCheckbox = (e) => {
    const { value, checked } = e.target;

    if (checked) {
      setFormData({
        ...formData,
        hobbies: [...formData.hobbies, value]
      });
    } else {
      setFormData({
        ...formData,
        hobbies: formData.hobbies.filter((hobby) => hobby !== value)
      });
    }
  };

  // Handle file upload
  const handleFileChange = (e) => {
    setFormData({
      ...formData,

import { useState, useEffect } from "react";

function LocalStorageUser() {

  const [name, setName] = useState("");
  const [savedUser, setSavedUser] = useState(null);

  const saveUser = () => {

    const user = { name };

    localStorage.setItem("user", JSON.stringify(user));

    setSavedUser(user);
  };

  useEffect(() => {

    const storedUser = JSON.parse(localStorage.getItem("user"));

    if (storedUser) {
      setSavedUser(storedUser);
    }

  }, []);

  return (
    <div>

      <h2>Local Storage Example</h2>

      <input
        type="text"
        placeholder="Enter Name"
        onChange={(e) => setName(e.target.value)}
      />

      <button onClick={saveUser}>Save</button>

      {savedUser && (
        <p>Stored User: {savedUser.name}</p>
      )}

    </div>
  );
}

// Function to fetch and display JSON
async function fetchJSON() {
    try {
        const response = await fetch('data.json');
        if (!response.ok) throw new Error("Network response was not ok");
        const data = await response.json();
        
        renderTable(data, ['Name', 'Age', 'City'], ['name', 'age', 'city']);
    } catch (error) {
        console.error("Error fetching JSON:", error);
    }
}

// Function to fetch and display XML
async function fetchXML() {
    try {
        const response = await fetch('data.xml');
        const text = await response.text();
        const parser = new DOMParser();
        const xmlDoc = parser.parseFromString(text, "text/xml");
        
        const students = xmlDoc.getElementsByTagName("student");
        let dataList = [];

        for (let i = 0; i < students.length; i++) {
            dataList.push({
                name: students[i].getElementsByTagName("name")[0].childNodes[0].nodeValue,
                age: students[i].getElementsByTagName("age")[0].childNodes[0].nodeValue
            });
        }
        
        renderTable(dataList, ['Name', 'Age'], ['name', 'age']);
    } catch (error) {
        console.error("Error fetching XML:", error);
    }
}

// Helper function to render the table dynamically
function renderTable(data, headers, keys) {
    const head = document.getElementById('table-head');
    const body = document.getElementById('table-body');
    
    // Clear existing content
    head.innerHTML = "";
    body.innerHTML = "";

    // Create Headers
    headers.forEach(text => {
        let th = document.createElement('th');
        th.innerText = text;
        head.appendChild(th);
    });

    // Create Rows
    data.forEach(item => {
        let row = body.insertRow();
        keys.forEach(key => {
            let cell = row.insertCell();
            cell.innerText = item[key];
        });
    });
}

import React, { useState } from 'react';

function LoginForm() {

  // useState stores current value of each field
  // [value, setterFunction] = useState(initialValue)
  const [username, setUsername] = useState('');
  const [email,    setEmail]    = useState('');
  const [password, setPassword] = useState('');

  // errors object stores error messages for each field
  const [errors, setErrors] = useState({});

  // ---- VALIDATION FUNCTION ----
  function validate() {
    let newErrors = {};

    // Rule 1: Username must not be empty
    if (username.trim() === '') {
      newErrors.username = 'Username cannot be empty.';
    }

    // Rule 2: Email must contain @ and a domain (e.g. user@example.com)
    // The regex pattern checks for valid email format
    const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailPattern.test(email)) {
      newErrors.email = 'Please enter a valid email (e.g. user@example.com).';
    }

    // Rule 3: Password validations
    const hasCapital   = /[A-Z]/.test(password);       // at least 1 uppercase letter
    const hasSpecial   = /[!@#$%^&*]/.test(password);  // at least 1 special character
    const validLength  = password.length >= 6 && password.length <= 12;

    if (!validLength) {
      newErrors.password = 'Password must be 6 to 12 characters long.';
    } else if (!hasCapital) {
      newErrors.password = 'Password must contain at least one capital letter.';
    } else if (!hasSpecial) {
      newErrors.password = 'Password must contain at least one special character (!@#$%^&*).';
    }

    return newErrors;
  }

  // ---- SUBMIT HANDLER ----
  function handleSubmit(e) {
    e.preventDefault(); // Prevents page from refreshing on submit

    const foundErrors = validate();

    if (Object.keys(foundErrors).length > 0) {
      // There are errors - show them on screen
      setErrors(foundErrors);
    } else {
      // No errors - form is valid!
      setErrors({});
      alert('Login Successful! Welcome, ' + username + '!');
    }
  }

  // ---- RENDER ----
  return (
    <div style={{ maxWidth: '420px', margin: '40px auto', fontFamily: 'Arial' }}>
      <h2>Login Page</h2>

      <form onSubmit={handleSubmit}>

        {/* Username Field */}
        <div style={{ marginBottom: '16px' }}>
          <label>Username:</label><br />
          <input
            type="text"
            value={username}
            onChange={(e) => setUsername(e.target.value)}
            placeholder="Enter username"
            style={{ width: '100%', padding: '8px', marginTop: '4px' }}
          />
          {errors.username && (
            <p style={{ color: 'red', margin: '4px 0' }}>{errors.username}</p>
          )}
        </div>

        {/* Email Field */}
        <div style={{ marginBottom: '16px' }}>
          <label>Email:</label><br />
          <input
            type="text"
            value={email}
            onChange={(e) => setEmail(e.target.value)}
            placeholder="Enter email"
            style={{ width: '100%', padding: '8px', marginTop: '4px' }}
          />
          {errors.email && (
            <p style={{ color: 'red', margin: '4px 0' }}>{errors.email}</p>
          )}
        </div>

        {/* Password Field */}
        <div style={{ marginBottom: '16px' }}>
          <label>Password:</label><br />
          <input
            type="password"
            value={password}
            onChange={(e) => setPassword(e.target.value)}
            placeholder="Enter password (6-12 chars)"
            style={{ width: '100%', padding: '8px', marginTop: '4px' }}
          />
          {errors.password && (
            <p style={{ color: 'red', margin: '4px 0' }}>{errors.password}</p>
          )}
        </div>

        <button
          type="submit"
          style={{ padding: '10px 24px', cursor: 'pointer', fontSize: '16px' }}
        >
          Login
        </button>

      </form>
    </div>
  );
}

import React, { useState, useEffect } from 'react';

function Comments() {

  // comments: stores the fetched comments array (starts empty)
  const [comments, setComments] = useState([]);

  // loading: true while data is being fetched
  const [loading, setLoading] = useState(true);

  // useEffect runs once when the component mounts (appears on screen)
  // The [] at the end means "run only once, not on every re-render"
  useEffect(() => {

    // fetch() makes an AJAX request - loads data without refreshing page
    fetch('/comments.json')
      .then(function(response) {
        return response.json(); // convert raw response to JavaScript object
      })
      .then(function(data) {
        setComments(data);   // save fetched comments into state
        setLoading(false);   // done loading
      })
      .catch(function(error) {
        console.error('Error loading comments:', error);
        setLoading(false);
      });

  }, []); // <-- empty array = run only once on page load

  // Show loading message while data is being fetched
  if (loading) {
    return <p>Loading comments, please wait...</p>;
  }

  // Once loaded, display the comments
  return (
    <div style={{ maxWidth: '600px', margin: '20px auto', fontFamily: 'Arial' }}>
      <h2>User Comments ({comments.length} total)</h2>

      {comments.map(function(c) {
        return (
          <div
            key={c.id}
            style={{
              border: '1px solid #cccccc',
              borderRadius: '8px',
              padding: '14px',
              marginBottom: '12px',
              backgroundColor: '#f9f9f9'
            }}
          >
            <strong>{c.user}</strong>
            <p style={{ margin: '6px 0 0' }}>{c.comment}</p>
          </div>
        );
      })}

import React, { useState, useEffect } from 'react';

function Movies() {

  // movies: stores the array of movie objects (starts empty)
  const [movies, setMovies] = useState([]);

  // loading: shows a message while data is being fetched
  const [loading, setLoading] = useState(true);

  // useEffect with [] = runs once when component first loads
  useEffect(() => {

    // AJAX call using fetch() - loads movies.json without refreshing
    fetch('/movies.json')
      .then(function(response) {
        return response.json(); // parse response as JSON
      })
      .then(function(data) {
        setMovies(data);   // save movies into state
        setLoading(false); // stop showing loading message
      })
      .catch(function(error) {
        console.error('Error fetching movies:', error);
        setLoading(false);
      });

  }, []);

  // Show loading message while fetching
  if (loading) {
    return <p>Loading movies...</p>;
  }

  // Table styles
  const tableStyle = {
    width: '100%',
    borderCollapse: 'collapse',
    fontFamily: 'Arial'
  };

  const thStyle = {
    backgroundColor: '#333333',
    color: 'white',
    padding: '12px',
    textAlign: 'left'
  };

  const tdStyle = {
    padding: '10px 12px',
    borderBottom: '1px solid #dddddd'
  };

  return (
    <div style={{ maxWidth: '700px', margin: '30px auto', fontFamily: 'Arial' }}>
      <h2>Now Showing - Available Movies</h2>

      <table style={tableStyle}>
        <thead>
          <tr>
            <th style={thStyle}>#</th>
            <th style={thStyle}>Movie Title</th>
            <th style={thStyle}>Rating</th>
            <th style={thStyle}>Duration</th>
          </tr>
        </thead>
        <tbody>
          {movies.map(function(movie, index) {
            return (
              <tr
                key={movie.id}
                style={{ backgroundColor: index % 2 === 0 ? '#f5f5f5' : 'white' }}
              >
                <td style={tdStyle}>{index + 1}</td>
                <td style={tdStyle}>{movie.title}</td>
                <td style={tdStyle}>&#11088; {movie.rating} / 10</td>
                <td style={tdStyle}>{movie.duration}</td>
              </tr>
            );
          })}
        </tbody>
      </table>

      <p style={{ marginTop: '12px' }}>
        Total movies available: <strong>{movies.length}</strong>
      </p>
    </div>
  );
}

export default Movies;

Q4(a)  Node.js - Append Customer Feedback to feedback.txt
        [Marks: as per paper]
        Task: Create a Node.js program using the fs module to append
              customer feedback to a feedback.txt file.
================================================================================

CONCEPT:
  - fs.appendFileSync() adds text to the END of a file (never deletes old data)
  - Different from writeFileSync() which REPLACES/overwrites the file
  - appendFileSync() creates the file automatically if it doesn't exist

--------------------------------------------------------------------------------
FILE: feedback.js
--------------------------------------------------------------------------------

// Require the fs (File System) module - built into Node.js, no install needed
const fs = require('fs');

const filename = 'feedback.txt';

// ---- FUNCTION TO APPEND ONE FEEDBACK ENTRY ----
function appendFeedback(customerName, feedbackText) {
  // Get current date and time
  const timestamp = new Date().toLocaleString();

  // Format the feedback entry
  const entry = '[' + timestamp + '] ' + customerName + ': ' + feedbackText + '\n';

  // appendFileSync adds to END of file (creates file if it does not exist)
  fs.appendFileSync(filename, entry);

  console.log('Feedback saved for: ' + customerName);
}

// ---- SAMPLE CUSTOMER FEEDBACKS ----
console.log('--- Saving Customer Feedback ---\n');

appendFeedback('Rahul Sharma',  'Great service! Very satisfied with the product.');
appendFeedback('Priya Patel',   'Delivery was fast. Will order again!');
appendFeedback('Amit Kumar',    'Product quality could be improved.');
appendFeedback('Sneha Joshi',   'Excellent customer support team.');

console.log('\nAll feedback saved to ' + filename);

// ---- READ AND DISPLAY FULL FILE CONTENT ----
console.log('\n--- Current Contents of ' + filename + ' ---');
const content = fs.readFileSync(filename, 'utf8');
console.log(content);
console.log('--- End of File ---');
